
sprintf() 函数的使用，sprintf(xxx, "%s", value); 将内容以字符串的形式赋值给相关变量

```c
#include <stdio.h>
#include <stdlib.h>

struct Student
{
   char num[100];
   char name[20];
   int age;
};

int main()
{
   struct Student stus[3];
   sprintf(stus[0].num, "%s", "A10001");
   sprintf(stus[0].name, "%s", "zhangyz");

   return 0;
}
```
