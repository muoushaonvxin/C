#include <stdio.h>
#include <ctype.h>

void ToUpper(const char * pstr){
    while(*pstr){ // \0字符的ASCII码值为0
        printf("%c", toupper(*pstr));
        ++pstr;
    }
}

int main(){
    const char *pstr = "hello world";
    void(*pFunc)(); // 先定义
    pFunc = ToUpper; // 在赋值
    (*pFunc)(pstr); // 通过函数指针进行调用
    return 0;
}
