```c
#include <stdio.h>

int main(){
    int a = 3, b = 4;
    
    //通过一个函数来交换变量a和b的值
    swap(a,b);
    printf("a = %d, b = %d\n", a, b);
    return 0;
}

void swap(int a, int b){
    int t;
    t = a;
    a = b;
    b = t;
}
```
